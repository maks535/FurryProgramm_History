using System;
using System.IO;
using System.Diagnostics;
using System.Linq;

class Program
{
    static void Main()
    {
        string folderPath = GetOrCreateFolderPath();
        if (!Directory.Exists(folderPath))
        {
            Console.WriteLine("Папка не найдена.");
            return;
        }

        string[] files = Directory.GetFiles(folderPath);

        while (true)
        {
            Console.Write("Введите имена файлов через ':', или пустой ввод для списка: ");
            string input = Console.ReadLine();

            if (string.IsNullOrWhiteSpace(input))
            {
                Console.WriteLine("Доступные файлы:");
                for (int i = 0; i < files.Length; i++)
                    Console.WriteLine($"[{i + 1}] {Path.GetFileName(files[i])}");

                continue;
            }

            string[] requestedFiles = input.Split(':');

            foreach (string fileName in requestedFiles)
            {
                string[] matches = files.Where(f => Path.GetFileName(f).ToLower().Contains(fileName.ToLower().Trim())).ToArray();

                if (matches.Length == 1)
                    LaunchFile(matches[0]);
                else if (matches.Length > 1)
                {
                    Console.WriteLine($"Найдено несколько файлов для \"{fileName}\":");
                    for (int i = 0; i < matches.Length; i++)
                        Console.WriteLine($"[{i + 1}] {Path.GetFileName(matches[i])}");

                    Console.Write("Выберите номер файла: ");
                    if (int.TryParse(Console.ReadLine(), out int choice) && choice > 0 && choice <= matches.Length)
                        LaunchFile(matches[choice - 1]);
                }
                else
                    Console.WriteLine($"Файл \"{fileName}\" не найден.");
            }
        }
    }

    static string GetOrCreateFolderPath()
    {
        const string configFile = "config.txt";
        if (File.Exists(configFile)) return File.ReadAllText(configFile).Trim();
        Console.Write("Введите путь к папке: ");
        string folderPath = Console.ReadLine();
        File.WriteAllText(configFile, folderPath);
        return folderPath;
    }

    static void LaunchFile(string filePath)
    {
        Console.WriteLine($"Запуск {Path.GetFileName(filePath)}...");
        try { Process.Start(filePath); }
        catch (Exception ex) { Console.WriteLine($"Ошибка: {ex.Message}"); }
    }
}
